SELECT AGB_CODE AS AGENT_CODE, AG.AGB_NAME1 AS AGENT_NAME,
                AG.AGB_ADD1 || ' ' || AG.AGB_ADD2 || ' ' || AG.AGB_ADD3 || ' ' || AG.AGB_ADD4 || ' ' || AG.AGB_ADD5 || ' ' || AG.AGB_ADD6 AS AGENT_ADDRESS,
                PT.PRT_NAME AS CITY, CO.COU_NAME AS COUNTRY, CA.CA_AREA AS AREA

                FROM APLCNT.TAB_AGB AG
                LEFT OUTER JOIN TAB_PRT PT ON SUBSTR(AG.AGB_CODE,1,5)=PT.PRT_CODE
                LEFT OUTER JOIN TAB_COUNTRY CO ON SUBSTR(AG.AGB_CODE,1,2)=CO.COU_ISO2
                LEFT OUTER JOIN TAB_COU_AREA CA ON CO.COU_ISO3=CA.CA_COU
